# Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the existing PropIntel single-property residential dashboard (`app/page.jsx`, ~720 lines) into a multi-mode analyzer: Residential ↔ Commercial mode toggle, side-by-side comparison of 2–4 properties, Recharts visualizations (projection line, sensitivity heatmap, score-breakdown bar), `localStorage` saved deals, and `zod` validation at the form boundary. Split the monolithic page into focused feature-folder components without rewriting the visual language.

**Architecture:** Feature-folder component organization under `app/components/`. `app/page.jsx` becomes a thin layout shell that composes feature components and owns top-level mode/compare/saved-deal state. Pure UI components receive props; data shaping and calc consumption stay in hooks (`app/hooks/`) and small lib modules (`app/lib/`). The calc engine (`lib/` from subsystem 1) and `/api/neighborhood` (rewired in subsystem 2) are consumed, never modified here.

**Tech Stack:** Next 16, React 19, Tailwind 3.4, framer-motion 12, lucide-react, clsx + tailwind-merge. Adds: `recharts`, `zod`. Dev/test: `vitest`, `@testing-library/react`, `@testing-library/jest-dom`, `jsdom`. Follow web coding-style: PascalCase components, kebab-case CSS class names, semantic HTML, compositor-friendly animation (transform/opacity only).

  ```js
  import { z } from 'zod';

  const pct = (label) => z.number({ invalid_type_error: `${label} must be a number` }).min(0).max(100);

  export const residentialSchema = z.object({
    purchasePrice: z.number().positive('Purchase price must be greater than 0'),
    repairCosts: z.number().min(0).default(0),
    downPaymentPct: pct('Down payment'),
    interestRate: z.number().min(0, 'Interest rate must be 0 or more').max(30),
    loanTermYears: z.number().int().positive(),
    annualPropertyTax: z.number().min(0),
    annualInsurance: z.number().min(0),
    monthlyHOA: z.number().min(0).default(0),
    monthlyRent: z.number().positive('Monthly rent must be greater than 0'),
    vacancyPct: pct('Vacancy'),
    managementPct: pct('Management fee'),
    maintenancePct: pct('Maintenance'),
    capExPct: pct('CapEx'),
    zipCode: z.string().regex(/^\d{5}$/, 'Zip code must be 5 digits').or(z.literal('')),
    bedrooms: z.number().int().min(0).max(6),
  });

  export const commercialSchema = z.object({
    purchasePrice: z.number().positive('Purchase price must be greater than 0'),
    squareFeet: z.number().positive('Square feet must be greater than 0'),
    units: z.number().int().min(1, 'Units must be at least 1'),
    annualGrossIncome: z.number().positive('Annual gross income must be greater than 0'),
    annualOperatingExpenses: z.number().min(0),
    downPaymentPct: pct('Down payment'),
    interestRate: z.number().min(0).max(30),
    loanTermYears: z.number().int().positive(),
    goingInCapRate: z.number().min(0).max(30),
    exitCapRate: z.number().min(0).max(30),
  });

  const SCHEMAS = { residential: residentialSchema, commercial: commercialSchema };

  export function validateDeal(mode, inputs) {
    const result = SCHEMAS[mode].safeParse(inputs);
    if (result.success) return { success: true, data: result.data, errors: {} };
    const errors = {};
    for (const issue of result.error.issues) {
      const key = issue.path[0];
      if (key && !errors[key]) errors[key] = issue.message;
    }
    return { success: false, data: null, errors };
  }
  ```

  ```js
  export const STORAGE_KEY = 'propintel.savedDeals.v1';

  export function loadDeals() {
    if (typeof window === 'undefined') return [];
    try {
      const raw = window.localStorage.getItem(STORAGE_KEY);
      const parsed = raw ? JSON.parse(raw) : [];
      return Array.isArray(parsed) ? parsed : [];
    } catch {
      return [];
    }
  }

  function persist(deals) {
    window.localStorage.setItem(STORAGE_KEY, JSON.stringify(deals));
    return deals;
  }

  export function saveDeal(deal) {
    const saved = {
      id: deal.id ?? `deal_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,
      name: deal.name,
      mode: deal.mode,
      inputs: deal.inputs,
      savedAt: Date.now(),
    };
    const existing = loadDeals().filter((d) => d.id !== saved.id);
    persist([...existing, saved]);
    return saved;
  }

  export function deleteDeal(id) {
    const remaining = loadDeals().filter((d) => d.id !== id);
    return persist(remaining);
  }
  ```

  ```js
  // Metric keys match lib/calculations.js result object (page.jsx:112-143).
  // higherIsBetter: false means a lower value wins (e.g. GRM).
  export const COMPARE_METRICS = [
    { key: 'investmentScore', label: 'Score', higherIsBetter: true },
    { key: 'monthlyCashFlow', label: 'Cash Flow', higherIsBetter: true },
    { key: 'cashOnCash', label: 'CoC', higherIsBetter: true },
    { key: 'capRate', label: 'Cap Rate', higherIsBetter: true },
    { key: 'annualROI', label: 'ROI', higherIsBetter: true },
    { key: 'GRM', label: 'GRM', higherIsBetter: false },
  ];

  export function rankDeals(deals) {
    const winners = {};
    for (const metric of COMPARE_METRICS) {
      let bestIdx = -1;
      let bestVal = null;
      deals.forEach((deal, idx) => {
        const v = deal.results?.[metric.key];
        if (v === null || v === undefined || Number.isNaN(v)) return;
        if (bestVal === null || (metric.higherIsBetter ? v > bestVal : v < bestVal)) {
          bestVal = v;
          bestIdx = idx;
        }
      });
      winners[metric.key] = bestIdx;
    }
    const ordered = [...deals].sort(
      (a, b) => (b.results?.investmentScore ?? 0) - (a.results?.investmentScore ?? 0),
    );
    return { ordered, winners };
  }
  ```

  ```jsx
  'use client';
  import { cn } from '../../lib/cn';

  const RATE_DELTAS = [-1, -0.5, 0, 0.5, 1];   // interest-rate points
  const CAP_DELTAS = [-1, -0.5, 0, 0.5, 1];    // exit/going-in cap points

  export function buildSensitivityGrid({ baseInputs, compute }) {
    return RATE_DELTAS.map((rateDelta) => ({
      rateDelta,
      cells: CAP_DELTAS.map((capDelta) => {
        const inputs = {
          ...baseInputs,
          interestRate: (baseInputs.interestRate ?? 0) + rateDelta,
          exitCapRate: (baseInputs.exitCapRate ?? baseInputs.goingInCapRate ?? 0) + capDelta,
        };
        return { capDelta, value: compute(inputs) };
      }),
    }));
  }

  const heatColor = (value, min, max) => {
    if (max === min) return 'bg-slate-100';
    const t = (value - min) / (max - min);
    if (t >= 0.66) return 'bg-emerald-200 text-emerald-900';
    if (t >= 0.33) return 'bg-amber-200 text-amber-900';
    return 'bg-rose-200 text-rose-900';
  };

  export const SensitivityHeatmap = ({ baseInputs, compute }) => {
    const grid = buildSensitivityGrid({ baseInputs, compute });
    const values = grid.flatMap((r) => r.cells.map((c) => c.value));
    const min = Math.min(...values);
    const max = Math.max(...values);
    return (
      <table className="w-full text-[10px] border-separate border-spacing-1" aria-label="Sensitivity: interest rate vs cap rate">
        <thead>
          <tr>
            <th className="text-slate-400 font-bold">Rate \ Cap</th>
            {CAP_DELTAS.map((c) => (
              <th key={c} className="text-slate-400 font-bold tabular-nums">{c > 0 ? `+${c}` : c}</th>
            ))}
          </tr>
        </thead>
        <tbody>
          {grid.map((row) => (
            <tr key={row.rateDelta}>
              <th className="text-slate-400 font-bold tabular-nums">{row.rateDelta > 0 ? `+${row.rateDelta}` : row.rateDelta}</th>
              {row.cells.map((cell) => (
                <td
                  key={cell.capDelta}
                  className={cn('text-center py-2 rounded font-bold tabular-nums', heatColor(cell.value, min, max))}
                >
                  {Math.round(cell.value)}
                </td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
    );
  };
  ```
